CREATE DATABASE Zongolica;
USE Zongolica;

CREATE TABLE Usuarios(
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(80),
    correo VARCHAR(120),
    rol VARCHAR(80),
    PRIMARY KEY (id)
    );
    
CREATE TABLE Notificaciones(
    id INT NOT NULL AUTO_INCREMENT,
    mensaje VARCHAR(200),
    fecha DATE,
    leido TINYINT,
    PRIMARY KEY (id)
    );
    
   CREATE TABLE Materias(
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(80),
    docente VARCHAR(120),
    grado TINYINT,
    PRIMARY KEY (id)
    );
    
    CREATE TABLE Alumnos(
    id INT NOT NULL AUTO_INCREMENT,
    idMateria INT NOT NULL,
    nombre VARCHAR(80),
    matricula VARCHAR(20),
    PRIMARY KEY (id),
        FOREIGN key (idMateria) REFERENCES Materias(id)
    );

CREATE TABLE Asistencias(
    id INT NOT NULL AUTO_INCREMENT,
    idMateria INT NOT NULL,
    idAlumno INT NOT NULL,
    estado VARCHAR(80),
    fecha DATE,
    hora TIME,
 	PRIMARY KEY (id),
    FOREIGN key (idMateria) REFERENCES Materias(id),
    FOREIGN key (idAlumno) REFERENCES Alumnos(id)
    );
